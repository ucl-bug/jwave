58 def spatial_axis(n, delta): 

59 if n % 2 == 0: 

60 return jnp.arange(0, n) * delta - delta * n / 2 

61 else: 

62 return jnp.arange(0, n) * delta - delta * (n - 1) / 2 

63 

64 axis = [spatial_axis(n, delta) for n, delta in zip(N, dx)] 

65 

66 k_axis = lambda n, d: jnp.fft.fftfreq(n, d) *2 * jnp.pi 

67 k_vec = [k_axis(n, delta) for n, delta in zip(N, dx)] 

68 cell_area = reduce(lambda x,y : x*y, dx) 

69 

70 return kGrid( 

71 N=N, 

72 dx=dx, 

73 k_vec=k_vec, 

74 cell_area=cell_area, 

75 k_staggered=None, 

76 k_with_kspaceop=None, 

77 space_axis=axis, 

78 )

80 def to_staggered(self): 

81 r"""Produces a copy of the grid with the staggered vectors field set 

82 

83 !!! example 

84 ```python 

85 grid = kGrid.make_grid(N=(64,64), dx=(0.3, 0.3)) 

86 grid = grid.to_staggered() 

87 ``` 

88 

89 """ 

90 tuple_to_dict = self._asdict() 

91 tuple_to_dict["k_staggered"] = { 

92 "backward": list( 

93 map( 

94 lambda x: x[0] * jnp.exp(1j * x[0] * x[1] / 2), 

95 zip(self.k_vec, self.dx), 

96 ) 

97 ), 

98 "forward": list( 

99 map( 

100 lambda x: x[0] * jnp.exp(-1j * x[0] * x[1] / 2), 

101 zip(self.k_vec, self.dx), 

102 ) 

103 ), 

104 } 

105 return kGrid(**tuple_to_dict)

107 def apply_kspace_operator(self, c_ref, dt): 

108 r"""Modifies the k-vectors used for derivative calculation 

109 by applying the k-space operator (see [k-Wave](www.k-wave.org)) 

110 

111 !!! warning 

112 It makes sense to use those vectors **only to calculate 

113 first order derivatives**. This is a bad design choice, 

114 but makes things faster than calculating the operator 

115 each time. 

116 

117 Args: 

118 c_ref (float): Reference speed of sound 

119 dt (float): Temporal stepsize 

120 

121 Returns: 

122 grid: An updated grid containing the modified k-vectors under the 

123 key `"kgrid_with_kspace_op"` 

124 

125 !!! example 

126 ```python 

127 grid = kGrid.make_grid(N=(64,64), dx=(0.3, 0.3)) 

128 grid = grid.to_staggered() 

129 grid = grid.apply_kspace_operator(c_ref = 1480, dt = time_array.dt) 

130 ``` 

131 

132 """ 

133 assert self.k_staggered 

134 tuple_to_dict = self._asdict() 

135 

136 K = jnp.stack(jnp.meshgrid(*self.k_vec, indexing="ij")) 

137 k_magnitude = jnp.sqrt(jnp.sum(K ** 2, 0)) 

138 

139 # TODO: Check why it seems to work better without k_space_op 

140 k_space_op = 1.0 # safe_sinc(c_ref * k_magnitude * dt/(2*jnp.pi)) 

141 modified_kgrid = jax.tree_util.tree_map(lambda x: x * k_space_op, K) 

142 

143 # Making staggered versions 

144 K = jnp.stack(jnp.meshgrid(*self.k_staggered["backward"], indexing="ij")) 

145 modified_kgrid_backward = jax.tree_util.tree_map(lambda x: x * k_space_op, K) 

146 K = jnp.stack(jnp.meshgrid(*self.k_staggered["forward"], indexing="ij")) 

147 modified_kgrid_forward = jax.tree_util.tree_map(lambda x: x * k_space_op, K) 

148 

149 # Update grid 

150 tuple_to_dict["k_with_kspaceop"] = { 

151 "plain": modified_kgrid, 

152 "backward": modified_kgrid_backward, 

153 "forward": modified_kgrid_forward, 

154 } 

155 return kGrid(**tuple_to_dict)

187def _points_on_circle(n, radius, centre, cast_int=True, angle=0.0): 

188 angles = np.linspace(0, 2 * np.pi, n, endpoint=False) 

189 x = (radius * np.cos(angles + angle) + centre[0]).tolist() 

190 y = (radius * np.sin(angles + angle) + centre[1]).tolist() 

191 if cast_int: 

192 x = list(map(int, x)) 

193 y = list(map(int, y)) 

194 return x, y

197def _circ_mask(N, radius, centre): 

198 x, y = np.mgrid[0 : N[0], 0 : N[1]] 

199 dist_from_centre = np.sqrt((x - centre[0]) ** 2 + (y - centre[1]) ** 2) 

200 mask = (dist_from_centre < radius).astype(int) 

201 return mask

287 @staticmethod 

288 def from_kgrid(grid: kGrid, medium: Medium, cfl: float = 0.3, t_end=None): 

289 r"""Construct a `TimeAxis` object from `kGrid` and `Medium` 

290 

291 Args: 

292 grid (kGrid): 

293 medium (Medium): 

294 cfl (float, optional): The [CFL number](http://www.k-wave.org/). Defaults to 0.3. 

295 t_end ([float], optional): The final simulation time. If None, 

296 it is automatically calculated as the time required to travel 

297 from one corner of the domain to the opposite one. 

298 

299 """ 

300 dt = dt = cfl * min(grid.dx) / jnp.max(medium.sound_speed) 

301 if t_end is None: 

302 t_end = jnp.sqrt( 

303 sum((x[-1] - x[0]) ** 2 for x in grid.space_axis) 

304 ) / jnp.min(medium.sound_speed) 

305 return TimeAxis(dt=dt, t_end=t_end)
